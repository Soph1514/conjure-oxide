24// The RuleResult struct represents the
25// result of applying a rule to an expression.
26#[derive(Debug, Clone)]
27pub struct RuleResult<'a> {
28    pub rule_data: RuleData<'a>,
29    pub reduction: Reduction,
30}
31
32/// Logs the application of a rule and its effects,
33/// to the main log, and the human readable traces used by the integration tester,
34/// that the rule has been applied to the expression
35pub fn log_rule_application(
36    result: &RuleResult,
37    initial_expression: &Expression,
38    initial_model: &SubModel,
39    consumer: &Option<Consumer>,
40) {
41    /// extracts data from the RuleResult struct
42    /// red = reduction and any constraints and variables
43    let red = &result.reduction;
44    let rule = result.rule_data.rule;
45    let new_top_string = pretty_vec(&red.new_top);
46
47    /// logs rule application to the main log
48    info!(
49        %new_top_string,
50        "Applying rule: {} ({:?}), to expression: {}, resulting in: {}",
51        rule.name,
52        rule.rule_sets,
53        initial_expression,
54        red.new_expression
55    );
56
57    // empty if no top level constraints
58    let top_level_str = if red.new_top.is_empty() {
59        String::new()
60    } else {
61        let mut exprs: Vec<String> = vec![];
62
63        for expr in &red.new_top {
64            exprs.push(format!("  {}", expr));
65        }
66
67        let exprs = exprs.iter().join("\n");
68
69        format!("new constraints:\n{}\n", exprs)
70    };
71
72    // empty if no new variables
73    // TODO: consider printing modified and removed declarations too, though removing a declaration in a rule is less likely.
74    let new_variables_str = {
75        let mut vars: Vec<String> = vec![];
76
77        for var_name in red.added_symbols(&initial_model.symbols()) {
78            #[allow(clippy::unwrap_used)]
79            vars.push(format!(
80                "  {}",
81                pretty_variable_declaration(&red.symbols, &var_name).unwrap()
82            ));
83        }
84        if vars.is_empty() {
85            String::new()
86        } else {
87            format!("new variables:\n{}", vars.join("\n"))
88        }
89    };
90
91    trace!(
92        target: "rule_engine_human",
93        "{}, \n   ~~> {} ({:?}) \n{} \n{}\n{}--\n",
94        initial_expression,
95        rule.name,
96        rule.rule_sets,
97        red.new_expression,
98        new_variables_str,
99        top_level_str
100    );
101
102    trace!(
103        target: "rule_engine",
104        "{}",
105    json!({
106        "rule_name": result.rule_data.rule.name,
107        "rule_priority": result.rule_data.priority,
108        "rule_set": {
109            "name": result.rule_data.rule_set.name,
110        },
111        "initial_expression": serde_json::to_value(initial_expression).unwrap(),
112        "transformed_expression": serde_json::to_value(&red.new_expression).unwrap()
113    })
114
115    );
116
117    if let Some(consumer) = consumer {
118        if check_verbosity_level(&consumer) != VerbosityLevel::Low {
119            let rule_trace = RuleTrace {
120                initial_expression: initial_expression.clone(),
121                rule_name: rule.name.to_string(),
122                rule_set_name: result.rule_data.rule_set.name.to_string(),
123                rule_priority: result.rule_data.priority,
124                transformed_expression: Some(red.new_expression.clone()),
125                new_variables_str: Some(new_variables_str.to_string()),
126                top_level_str: Some(top_level_str.to_string()),
127            };
128
129            capture_trace(&consumer, TraceStruct::RuleTrace(rule_trace));
130        }
131    }
132}

134/// Represents errors that can occur during the model rewriting process.
135#[derive(Debug, Error)]
136pub enum RewriteError {
137    #[error("Error resolving rules {0}")]
138    ResolveRulesError(ResolveRulesError),
139}
140
141impl From<ResolveRulesError> for RewriteError {
142    fn from(error: ResolveRulesError) -> Self {
143        RewriteError::ResolveRulesError(error)
144    }
